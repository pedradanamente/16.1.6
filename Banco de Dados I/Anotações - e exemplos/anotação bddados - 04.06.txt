EXEMPLO 

ALTER TABLE FISICA ADD CONSTRAINT FK_PESSOA FOREINGN KEY
	(ID) REFERENCES PESSOA(ID) ON DELETE CASCADE
	ON UPDATE CASCADE




INNER JOIN

SELECT PESSOA.ID AS IDENTIFICADOR,
PESSOA.NOME,
PESSOA.CPF,PEDIDO.DATA,
PP.QUANTIDADE, PP.QUANTIDADE, PP.VALOR_UNITARIO,
PRODUTO.DESCRICAO

FROM PESSOA
	INNER JOIN FISICA ON FISICA.ID = PESSOA.ID
	INNER JOIN JURIDICA ON JURIDICA.ID = PESSOA.ID
	INNER JOIN PEDIDO ON PEDIDO.ID_PESSOA = PESSOA.ID
	INNER JOIN PEDIDO_PRODUTO AS PP ON PP.ID_PEDIDO = PEDIDO.ID

--SCRIPT SEM INNER JOIN
SELECT PESSOA.NOME, FISICA.CPF, PEDIDO.DATA, PRODUTO.DESCRICAO
	PP.QUANTIDADE, PP.VALOR_UNITARIO, PRODUTO
	PP.QUANTIDADE * PP.VALOR_UNITARIO AS TOTAL

FROM PESSOA, FISICA, PEDIDO, PEDIDO_PRODUTO AS PP
WHERE
	PESSOA.ID = FISICA.ID
	AND PEDIDO.ID_PESSOA = PESSOA.ID
	AND PEDIDO.ID = PP.ID_PEDIDO
	AND PRODUTO.ID = PP.ID_PRODUTO

--SELECT AGRUPADOS (ANINAHDOS)
SELECT PESSOA.NOME, FISICA.CPF, PEDIDO.DATA
FROM PESSOA, FISICA
	WHERE PESSOA.ID IN (SELECT ID FROM FISICA)
	AND PESSOA.ID IN (SELECT ID_PESSOA FROM PEDIDO)
