CREATE PROCEDURE 'SP_CLIENTE_SALARIO'
AS
	DECLARE @MAX DECIMAL(10,2), @MIN DECIMAL(10,2)
	
	SET @MAX = SELECT TOP 1 MAX (SALARIO) FROM CLIENTE
	SET @MIN = SELECT TOP 2 MIN (SALARIO) FROM CLIENTE

	SELECT CLIENTE_NOME, SALARIO FROM CLIENTE
		WHERE SALARIO = @MAX OR 
			SALARIO = @MIN

CREATE TRIGGER 'TG_CLIENTE' ON CLIENTE AFTER INSERT
AS

	DECLARE @SALARIO NUMERIC(10,2);
	DECLARE @ID_CLIENTE INT;

SELECT @ID_CLIENTE = CLIENTE_ID,
	@SALARIO = SALARIO FROM INSERTED

IF @SALARIO <= 500
	@SALARIO = @SALARIO *1.1
ELSE
	@SALARIO = @SALARIO * 1.05

UPDATE CLIENTE SET SALARIO = @SALARIO
	WHERE CLIENTE.ID = @ID_CLIENTE
