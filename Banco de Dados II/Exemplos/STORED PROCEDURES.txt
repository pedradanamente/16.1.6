STORED PROCEDURES

CREATE PROCEDURE SP_INSERTCLIENTES
@NOME VARCHAR(30),
@RG VARCHAR(20),
@CPF VARCHAR(11),
@ID_ESTADO_CIVIL INT

AS
IF @NOME<>"" AND @CPF<>"" AND @ID_ESTADO_CIVIL<>0
BEGIN
DECLARE @EC_ID INT
SELECT @EC_ID = ID FROM ESTADO_CIVIL WHERE ID= @ID_ESTADO_CIVIL
IF @EC_ID<>0

INSERT(NOME, CPF, RG, CC_ID) VALUES
	(@NOME, @CPF, @RG, @ID_ESTADO_CIVIL)


END
ELSE
PRINT 'MENSAGEM'
